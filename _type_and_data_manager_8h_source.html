366 TDest ConvertFloatToInt(TSource src)
367 {
368  TDest dest;
369  if (std::isnan(src)) {
370  dest = std::numeric_limits<TDest>::max();
371  } else if (std::isinf(src)) {
372  dest = src < 0 ? std::numeric_limits<TDest>::min() : std::numeric_limits<TDest>::max();
373  } else if (src < std::numeric_limits<TDest>::min()) {
374  dest = std::numeric_limits<TDest>::min();
375  } else if (src > std::numeric_limits<TDest>::max()) {
376  dest = std::numeric_limits<TDest>::max();
377  } else {
378  dest = static_cast<TDest>(RoundToEven(src));
379  }
380  return dest;
